here is the command that I used to find them...

find glassfish.* j2ee.sun.* -type f -exec egrep 'http://.*sun\.com/' {} \; | egrep -v '\<web\-app' | egrep -v '\<persistence' | egrep -v '\<faces\-config' | egrep -v '\<ejb\-jar' | egrep -v '\<application' | egrep -v '/dtds/.*\.dtd' | egrep -v '/xml/ns/.*\.xsd' | sort | uniq
